#include <Arduino.h>
#include <Ps3Controller.h>

int player = 0;
int battery = 0;
Ps3Controller p1;
Ps3Controller p2;

void notify(Ps3Controller ctrlr)
{
    //--- Digital cross/square/triangle/circle button events ---
    if( ctrlr.event.button_down.cross )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the cross button");
    if( ctrlr.event.button_up.cross )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the cross button");

    if( ctrlr.event.button_down.square )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the square button");
    if( ctrlr.event.button_up.square )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the square button");

    if( ctrlr.event.button_down.triangle )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the triangle button");
    if( ctrlr.event.button_up.triangle )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the triangle button");

    if( ctrlr.event.button_down.circle )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the circle button");
    if( ctrlr.event.button_up.circle )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the circle button");

    //--------------- Digital D-pad button events --------------
    if( ctrlr.event.button_down.up )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the up button");
    if( ctrlr.event.button_up.up )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the up button");

    if( ctrlr.event.button_down.right )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the right button");
    if( ctrlr.event.button_up.right )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the right button");

    if( ctrlr.event.button_down.down )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the down button");
    if( ctrlr.event.button_up.down )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the down button");

    if( ctrlr.event.button_down.left )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the left button");
    if( ctrlr.event.button_up.left )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the left button");

    //------------- Digital shoulder button events -------------
    if( ctrlr.event.button_down.l1 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the left shoulder button");
    if( ctrlr.event.button_up.l1 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the left shoulder button");

    if( ctrlr.event.button_down.r1 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the right shoulder button");
    if( ctrlr.event.button_up.r1 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the right shoulder button");

    //-------------- Digital trigger button events -------------
    if( ctrlr.event.button_down.l2 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the left trigger button");
    if( ctrlr.event.button_up.l2 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the left trigger button");

    if( ctrlr.event.button_down.r2 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the right trigger button");
    if( ctrlr.event.button_up.r2 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the right trigger button");

    //--------------- Digital stick button events --------------
    if( ctrlr.event.button_down.l3 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the left stick button");
    if( ctrlr.event.button_up.l3 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the left stick button");

    if( ctrlr.event.button_down.r3 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the right stick button");
    if( ctrlr.event.button_up.r3 )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the right stick button");

    //---------- Digital select/start/ps button events ---------
    if( ctrlr.event.button_down.select )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the select button");
    if( ctrlr.event.button_up.select )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the select button");

    if( ctrlr.event.button_down.start )
        Serial.println("Started pressing the start button");
    if( ctrlr.event.button_up.start )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the start button");

    if( ctrlr.event.button_down.ps )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Started pressing the Playstation button");
    if( ctrlr.event.button_up.ps )
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.println("Released the Playstation button");


    //---------------- Analog stick value events ---------------
   if( abs(ctrlr.event.analog_changed.stick.lx) + abs(ctrlr.event.analog_changed.stick.ly) > 2 ){
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Moved the left stick:");
       Serial.print(" x="); Serial.print(ctrlr.data.analog.stick.lx, DEC);
       Serial.print(" y="); Serial.print(ctrlr.data.analog.stick.ly, DEC);
       Serial.println();
    }

   if( abs(ctrlr.event.analog_changed.stick.rx) + abs(ctrlr.event.analog_changed.stick.ry) > 2 ){
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Moved the right stick:");
       Serial.print(" x="); Serial.print(ctrlr.data.analog.stick.rx, DEC);
       Serial.print(" y="); Serial.print(ctrlr.data.analog.stick.ry, DEC);
       Serial.println();
   }

   //--------------- Analog D-pad button events ----------------
   if( abs(ctrlr.event.analog_changed.button.up) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the up button: ");
       Serial.println(ctrlr.data.analog.button.up, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.right) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the right button: ");
       Serial.println(ctrlr.data.analog.button.right, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.down) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the down button: ");
       Serial.println(ctrlr.data.analog.button.down, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.left) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the left button: ");
       Serial.println(ctrlr.data.analog.button.left, DEC);
   }

   //---------- Analog shoulder/trigger button events ----------
   if( abs(ctrlr.event.analog_changed.button.l1)){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the left shoulder button: ");
       Serial.println(ctrlr.data.analog.button.l1, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.r1) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the right shoulder button: ");
       Serial.println(ctrlr.data.analog.button.r1, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.l2) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the left trigger button: ");
       Serial.println(ctrlr.data.analog.button.l2, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.r2) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the right trigger button: ");
       Serial.println(ctrlr.data.analog.button.r2, DEC);
   }

   //---- Analog cross/square/triangle/circle button events ----
   if( abs(ctrlr.event.analog_changed.button.triangle)){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the triangle button: ");
       Serial.println(ctrlr.data.analog.button.triangle, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.circle) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the circle button: ");
       Serial.println(ctrlr.data.analog.button.circle, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.cross) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the cross button: ");
       Serial.println(ctrlr.data.analog.button.cross, DEC);
   }

   if( abs(ctrlr.event.analog_changed.button.square) ){
       Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
       Serial.print("Pressing the square button: ");
       Serial.println(ctrlr.data.analog.button.square, DEC);
   }

   //---------------------- Battery events ---------------------
    if( battery != ctrlr.data.status.battery ){
        battery = ctrlr.data.status.battery;
        Serial.print("==");(ctrlr.getAddress());Serial.print("==");Serial.print("==");
        Serial.print("The controller battery is ");
        if( battery == ps3_status_battery_charging )      Serial.println("charging");
        else if( battery == ps3_status_battery_full )     Serial.println("FULL");
        else if( battery == ps3_status_battery_high )     Serial.println("HIGH");
        else if( battery == ps3_status_battery_low)       Serial.println("LOW");
        else if( battery == ps3_status_battery_dying )    Serial.println("DYING");
        else if( battery == ps3_status_battery_shutdown ) Serial.println("SHUTDOWN");
        else Serial.println("UNDEFINED");
    }

}

void notify1(){
    notify(Ps3);
}
void notify2(){
    notify(p2);
}

void onConnect(){
    Serial.println("Connected.");
}

void setup()
{
    Serial.begin(115200);

    
    Ps3.attach(notify1);
    Ps3.attachOnConnect(onConnect);
    Ps3.begin("00:00:00:00:00:01");

    // p2.attach(notify2);
    // p2.attachOnConnect(onConnect);
    // p2.begin("00:00:00:00:00:02");

    

    Serial.println("Ready.");
}
